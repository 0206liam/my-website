async function send() { const input = document.getElementById('user-input'); const text = input.value.trim(); if(!text) return; appendMsg(text, 'user'); input.value = "AI 運算中..."; input.disabled = true; try { // Step A: 先抓取最新的資料 (確保不會覆蓋掉別人的訊息) const res = await fetch(`https://api.github.com/gists/${GIST_ID}?t=${Date.now()}`, { headers: { "Authorization": `token ${USER_TOKEN}` } }); const data = await res.json(); // 檢查檔案是否存在 if (!data.files || !data.files["chats.json"]) { throw new Error("Gist 檔案名稱不對，請確認是 chats.json"); } let chats = JSON.parse(data.files["chats.json"].content || "[]"); // Step B: 處理用戶 ID (localStorage) let uid = localStorage.getItem('ai_uid'); if(!uid) { const users = [...new Set(chats.map(c => c.id))]; const count = users.length + 1; const random4 = Math.floor(1000 + Math.random() * 9000).toString(); uid = random4 + count.toString().padStart(3, '0'); localStorage.setItem('ai_uid', uid); } // Step C: 推送新訊息 chats.push({ id: uid, msg: text, time: new Date().toLocaleString(), role: 'user' }); // Step D: 【關鍵修正】正確的 PATCH 格式 await fetch(`https://api.github.com/gists/${GIST_ID}`, { method: "PATCH", headers: { "Authorization": `token ${USER_TOKEN}`, "Content-Type": "application/json" }, body: JSON.stringify({ files: { "chats.json": { content: JSON.stringify(chats, null, 2) // 必須包在 files -> 檔名 -> content 裡面 } } }) }); input.value = ""; input.disabled = false; } catch (e) { console.error(e); alert("連線失敗，請檢查授權碼或 Gist 設定"); input.disabled = false; input.value = ""; } }
